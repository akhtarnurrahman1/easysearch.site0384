    <script>
        // Global variables
        let currentSlideIndex = 0;
        let slides = [
            {
                content: `<h1>Welcome to PowerPoint Suite</h1>
                         <p>Click here to start editing your presentation...</p>
                         <ul>
                             <li>Create amazing slides</li>
                             <li>Add images and content</li>
                             <li>Present in fullscreen mode</li>
                         </ul>`,
                theme: 'default'
            }
        ];

        let themes = [
            { name: 'default', background: 'white', color: '#333' },
            { name: 'blue', background: 'linear-gradient(135deg, #667eea, #764ba2)', color: 'white' },
            { name: 'green', background: 'linear-gradient(135deg, #11998e, #38ef7d)', color: 'white' },
            { name: 'purple', background: 'linear-gradient(135deg, #a8edea, #fed6e3)', color: '#333' },
            { name: 'dark', background: '#1a202c', color: '#e2e8f0' }
        ];

        // Tab switching
        function switchTab(tabName) {
            document.querySelectorAll('.tab-content').forEach(tab => {
                tab.classList.remove('active');
            });
            document.querySelectorAll('.tab-button').forEach(btn => {
                btn.classList.remove('active');
            });
            
            document.getElementById(tabName).classList.add('active');
            event.target.classList.add('active');
            
            if (tabName === 'codeeditor') {
                updatePreview();
            }
        }

        // PowerPoint Functions
        function addSlide() {
            const newSlide = {
                content: `<h1>New Slide ${slides.length + 1}</h1><p>Edit this slide content...</p>`,
                theme: 'default'
            };
            slides.push(newSlide);
            updateSlidesList();
            selectSlide(slides.length - 1);
        }

        function deleteSlide() {
            if (slides.length > 1) {
                slides.splice(currentSlideIndex, 1);
                if (currentSlideIndex >= slides.length) {
                    currentSlideIndex = slides.length - 1;
                }
                updateSlidesList();
                selectSlide(currentSlideIndex);
            } else {
                alert('Cannot delete the last slide!');
            }
        }

        function selectSlide(index) {
            // Save current slide content
            if (slides[currentSlideIndex]) {
                slides[currentSlideIndex].content = document.getElementById('slide-content').innerHTML;
            }
            
            currentSlideIndex = index;
            document.getElementById('slide-content').innerHTML = slides[index].content;
            
            // Update active slide in list
            document.querySelectorAll('.slide-item').forEach((item, i) => {
                item.classList.toggle('active', i === index);
            });
            
            updateSlideCounter();
            applySlideTheme();
        }

        function updateSlidesList() {
            const slidesList = document.getElementById('slides-list');
            slidesList.innerHTML = '';
            
            slides.forEach((slide, index) => {
                const slideItem = document.createElement('div');
                slideItem.className = `slide-item ${index === currentSlideIndex ? 'active' : ''}`;
                slideItem.textContent = `Slide ${index + 1}`;
                slideItem.onclick = () => selectSlide(index);
                slidesList.appendChild(slideItem);
            });
            
            updateSlideCounter();
        }

        function updateSlideCounter() {
            document.getElementById('current-slide').textContent = currentSlideIndex + 1;
            document.getElementById('total-slides').textContent = slides.length;
        }

        function changeTheme() {
            const currentTheme = slides[currentSlideIndex].theme || 'default';
            const themeNames = themes.map(t => t.name);
            const currentIndex = themeNames.indexOf(currentTheme);
            const nextIndex = (currentIndex + 1) % themes.length;
            
            slides[currentSlideIndex].theme = themeNames[nextIndex];
            applySlideTheme();
        }

        function applySlideTheme() {
            const theme = themes.find(t => t.name === slides[currentSlideIndex].theme) || themes[0];
            const slideEditor = document.getElementById('slide-editor');
            slideEditor.style.background = theme.background;
            slideEditor.style.color = theme.color;
        }

        function insertImage() {
            const imageUrl = prompt('Enter image URL:');
            if (imageUrl) {
                const slideContent = document.getElementById('slide-content');
                slideContent.innerHTML += `<br><img src="${imageUrl}" style="max-width: 100%; height: auto; border-radius: 8px;" alt="Inserted image">`;
            }
        }

        function previousSlide() {
            if (currentSlideIndex > 0) {
                selectSlide(currentSlideIndex - 1);
            }
        }

        function nextSlide() {
            if (currentSlideIndex < slides.length - 1) {
                selectSlide(currentSlideIndex + 1);
            }
        }

        function startPresentation() {
            document.getElementById('fullscreen-presentation').style.display = 'flex';
            showFullscreenSlide(currentSlideIndex);
        }

        function showFullscreenSlide(index) {
            const theme = themes.find(t => t.name === slides[index].theme) || themes[0];
            const fullscreenContent = document.getElementById('fullscreen-slide-content');
            fullscreenContent.innerHTML = slides[index].content;
            fullscreenContent.style.background = theme.background;
            fullscreenContent.style.color = theme.color;
        }

        function previousSlideFullscreen() {
            if (currentSlideIndex > 0) {
                currentSlideIndex--;
                showFullscreenSlide(currentSlideIndex);
            }
        }

        function nextSlideFullscreen() {
            if (currentSlideIndex < slides.length - 1) {
                currentSlideIndex++;
                showFullscreenSlide(currentSlideIndex);
            }
        }

        function exitFullscreen() {
            document.getElementById('fullscreen-presentation').style.display = 'none';
        }

        // Code Editor Functions
        function updatePreview() {
            const code = document.getElementById('code-editor').value;
            const iframe = document.getElementById('preview-iframe');
            const iframeDoc = iframe.contentDocument || iframe.contentWindow.document;
            iframeDoc.open();
            iframeDoc.write(code);
            iframeDoc.close();
        }

        function changeLanguage() {
            const language = document.getElementById('language-select').value;
            const editor = document.getElementById('code-editor');
            
            switch(language) {
                case 'html':
                    editor.placeholder = '<!-- Write your HTML here -->';
                    break;
                case 'css':
                    editor.placeholder = '/* Write your CSS here */\nbody {\n    font-family: Arial, sans-serif;\n}';
                    break;
                case 'javascript':
                    editor.placeholder = '// Write your JavaScript here\nconsole.log("Hello World!");';
                    break;
            }
        }

        function newFile() {
            document.getElementById('code-editor').value = '';
            updatePreview();
        }

        function saveFile() {
            const content = document.getElementById('code-editor').value;
            const blob = new Blob([content], { type: 'text/plain' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'code-file.html';
            a.click();
            URL.revokeObjectURL(url);
        }

        function loadFile() {
            const input = document.createElement('input');
            input.type = 'file';
            input.accept = '.html,.css,.js,.txt';
            input.onchange = function(event) {
                const file = event.target.files[0];
                if (file) {
                    const reader = new FileReader();
                    reader.onload = function(e) {
                        document.getElementById('code-editor').value = e.target.result;
                        updatePreview();
                    };
                    reader.readAsText(file);
                }
            };
            input.click();
        }

        function formatCode() {
            const editor = document.getElementById('code-editor');
            let code = editor.value;
            
            // Simple HTML formatting
            code = code.replace(/></g, '>\n<');
            code = code.replace(/^\s+|\s+$/g, '');
            
            const lines = code.split('\n');
            let indentLevel = 0;
            const indentSize = 2;
            
            const formattedLines = lines.map(line => {
                line = line.trim();
                if (line.includes('</') && !line.includes('<!--')) {
                    indentLevel = Math.max(0, indentLevel - 1);
                }
                
                const formattedLine = ' '.repeat(indentLevel * indentSize) + line;
                
                if (line.includes('<') && !line.includes('</') && !line.includes('<!--') && !line.endsWith('/>')) {
                    indentLevel++;
                }
                
                return formattedLine;
            });
            
            editor.value = formattedLines.join('\n');
            updatePreview();
        }

        function refreshPreview() {
            updatePreview();
        }

        function fullscreenCode() {
            const editor = document.getElementById('code-editor');
            if (editor.requestFullscreen) {
                editor.requestFullscreen();
            }
        }

        // Event Listeners
        document.getElementById('code-editor').addEventListener('input', updatePreview);

        // Save slide content when editing
        document.getElementById('slide-content').addEventListener('input', function() {
            slides[currentSlideIndex].content = this.innerHTML;
        });

        // Keyboard shortcuts
        document.addEventListener('keydown', function(event) {
            if (event.ctrlKey || event.metaKey) {
                switch(event.key) {
                    case 's':
                        event.preventDefault();
                        if (document.getElementById('codeeditor').classList.contains('active')) {
                            saveFile();
                        }
                        break;
                    case 'n':
                        event.preventDefault();
                        if (document.getElementById('powerpoint').classList.contains('active')) {
                            addSlide();
                        } else {
                            newFile();
                        }
                        break;
                }
            }
            
            // Presentation navigation
            if (document.getElementById('fullscreen-presentation').style.display === 'flex') {
                switch(event.key) {
                    case 'ArrowLeft':
                        previousSlideFullscreen();
                        break;
                    case 'ArrowRight':
                        nextSlideFullscreen();
                        break;
                    case 'Escape':
                        exitFullscreen();
                        break;
                }
            }
        });

        // Initialize
        updateSlidesList();
        updatePreview();
        applySlideTheme();
    </script>
